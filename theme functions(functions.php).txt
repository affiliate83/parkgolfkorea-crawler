function parkgolf_setup() {
    load_theme_textdomain('parkgolf-korea', get_template_directory() . '/languages');
    add_theme_support('title-tag');
    add_theme_support('post-thumbnails');
    set_post_thumbnail_size(800, 500, true);
    add_image_size('parkgolf-card', 400, 250, true);
    add_image_size('parkgolf-wide', 1200, 600, true);
    add_theme_support('html5', array('search-form', 'comment-form', 'comment-list', 'gallery', 'caption', 'script', 'style'));
    add_theme_support('custom-logo', array('height' => 60, 'width' => 200, 'flex-height' => true, 'flex-width' => true));
    add_theme_support('automatic-feed-links');
    register_nav_menus(array(
        'primary' => '메인 네비게이션',
        'footer'  => '푸터 메뉴',
        'mobile'  => '모바일 메뉴',
    ));
}
add_action('after_setup_theme', 'parkgolf_setup');

function parkgolf_scripts() {
    $version = '1.0.2';

    wp_enqueue_style('google-fonts',
        'https://fonts.googleapis.com/css2?family=Noto+Sans+KR:wght@300;400;500;600;700;900&display=swap',
        array(), null);
    wp_enqueue_style('font-awesome',
        'https://cdn.jsdelivr.net/npm/@fortawesome/fontawesome-free@6.4.0/css/all.min.css',
        array(), '6.4.0');
    wp_enqueue_style('parkgolf-style',
        get_template_directory_uri() . '/assets/css/style.css',
        array('google-fonts', 'font-awesome'), $version);
    wp_enqueue_style('parkgolf-extra',
        get_template_directory_uri() . '/assets/css/theme-extra.css',
        array('parkgolf-style'), $version);

    if (is_page_template('page-courses.php') ||
        is_tax('region') ||
        is_singular('parkgolf_course')) {
        wp_enqueue_style('parkgolf-courses',
            get_template_directory_uri() . '/assets/css/courses.css',
            array('parkgolf-style'), $version);
    }

    wp_enqueue_script('parkgolf-main',
        get_template_directory_uri() . '/assets/js/main.js',
        array('jquery'), $version, true);

    if (is_page_template('page-courses.php') ||
        is_page('골프장-정보') ||
        is_tax('region')) {
        wp_enqueue_script('parkgolf-courses',
            get_template_directory_uri() . '/assets/js/courses.js',
            array('parkgolf-main'), $version, true);
    }

    if (is_singular() && comments_open() && get_option('thread_comments')) {
        wp_enqueue_script('comment-reply');
    }

    wp_localize_script('parkgolf-main', 'parkgolfData', array(
        'ajaxUrl'  => admin_url('admin-ajax.php'),
        'siteUrl'  => get_site_url(),
        'themeUrl' => get_template_directory_uri(),
        'nonce'    => wp_create_nonce('parkgolf_nonce'),
    ));
}
add_action('wp_enqueue_scripts', 'parkgolf_scripts');

function parkgolf_save_meta($post_id) {
    if (!isset($_POST['parkgolf_nonce']) && !isset($_POST['parkgolf_event_nonce'])) return;
    if (defined('DOING_AUTOSAVE') && DOING_AUTOSAVE) return;
    if (!current_user_can('edit_post', $post_id)) return;
    $all_fields = array(
        'course_address', 'course_holes', 'course_distance', 'course_hours',
        'course_fee', 'course_phone', 'course_reservation', 'course_reservation_url',
        'course_parking', 'course_facilities', 'course_map_lat', 'course_map_lng',
        'event_date_start', 'event_date_end', 'event_location', 'event_capacity',
        'event_fee', 'event_organizer', 'event_contact', 'event_apply_url',
        'event_status', 'event_prize',
    );
    foreach ($all_fields as $field) {
        if (isset($_POST[$field])) {
            update_post_meta($post_id, $field, sanitize_text_field($_POST[$field]));
        }
    }
}
add_action('save_post', 'parkgolf_save_meta');

function parkgolf_course_card($post_id) {
    $title       = get_the_title($post_id);
    $link        = get_permalink($post_id);
    $fee         = get_post_meta($post_id, 'course_fee', true);
    $reservation = get_post_meta($post_id, 'course_reservation', true);
    $address     = get_post_meta($post_id, 'course_address', true);
    $holes       = get_post_meta($post_id, 'course_holes', true);
    $hours       = get_post_meta($post_id, 'course_hours', true);

    $regions     = get_the_terms($post_id, 'region');
    $region_name = (!empty($regions) && !is_wp_error($regions)) ? $regions[0]->name : '';
    $region_slug = (!empty($regions) && !is_wp_error($regions)) ? $regions[0]->slug : '';

    $fee_label = (!empty($fee) && strpos($fee, '무료') !== false) ? '무료' : '유료';
    $fee_class = (!empty($fee) && strpos($fee, '무료') !== false) ? 'free' : 'paid';

    $thumbnail = get_the_post_thumbnail($post_id, 'parkgolf-card');
    ?>
    <div class="course-card" data-region="<?php echo esc_attr($region_slug); ?>">
        <a href="<?php echo esc_url($link); ?>" style="display:block;text-decoration:none;color:inherit;height:100%;">
            <div class="course-card-header">
                <span class="badge <?php echo esc_attr($fee_class); ?>"><?php echo esc_html($fee_label); ?></span>
                <?php if ($region_name): ?>
                    <span class="badge region"><?php echo esc_html($region_name); ?></span>
                <?php endif; ?>
                <?php if ($thumbnail): ?>
                    <?php echo $thumbnail; ?>
                <?php else: ?>
                    <div class="course-img-placeholder">⛳</div>
                <?php endif; ?>
            </div>
            <div class="course-card-body">
                <h3 class="course-title"><?php echo esc_html($title); ?></h3>
                <?php if ($address): ?>
                    <p class="course-address">
                        <i class="fas fa-map-marker-alt"></i> <?php echo esc_html($address); ?>
                    </p>
                <?php endif; ?>
                <div class="course-specs">
                    <?php if ($holes): ?>
                        <span><i class="fas fa-golf-ball"></i> <?php echo esc_html($holes); ?>홀</span>
                    <?php endif; ?>
                    <?php if ($fee): ?>
                        <span><i class="fas fa-won-sign"></i> <?php echo esc_html($fee_label); ?></span>
                    <?php endif; ?>
                    <?php if ($hours): ?>
                        <span><i class="fas fa-clock"></i> <?php echo esc_html($hours); ?></span>
                    <?php endif; ?>
                </div>
                <?php if ($reservation): ?>
                    <div class="course-reservation">
                        <span class="res-label">예약방법</span>
                        <span class="res-value"><?php echo esc_html($reservation); ?></span>
                    </div>
                <?php endif; ?>
            </div>
        </a>
    </div>
    <?php
}

// 7. 헬퍼 함수 - 이벤트 카드
// ============================================================
function parkgolf_event_card($post_id = null) {
    if (!$post_id) $post_id = get_the_ID();
    $title      = get_the_title($post_id);
    $link       = get_permalink($post_id);
    $date_start = get_post_meta($post_id, 'event_date_start', true);
    $location   = get_post_meta($post_id, 'event_location', true);
    $capacity   = get_post_meta($post_id, 'event_capacity', true);
    $apply_url  = get_post_meta($post_id, 'event_apply_url', true);
    $status     = get_post_meta($post_id, 'event_status', true) ?: '예정';

    $month = $date_start ? date('m월', strtotime($date_start)) : '';
    $day   = $date_start ? date('d',   strtotime($date_start)) : '';
    $year  = $date_start ? date('Y',   strtotime($date_start)) : '';

    $status_class = array('접수중' => 'badge-open', '마감' => 'badge-closed', '예정' => 'badge-soon');
    $badge_class  = $status_class[$status] ?? 'badge-soon';
    ?>
    <div class="schedule-item">
        <div class="schedule-date">
            <span class="month"><?php echo esc_html($month); ?></span>
            <span class="day"><?php echo esc_html($day); ?></span>
            <span class="year"><?php echo esc_html($year); ?></span>
        </div>
        <div class="schedule-info">
            <h3><a href="<?php echo esc_url($link); ?>"><?php echo esc_html($title); ?></a></h3>
            <div class="schedule-meta">
                <?php if ($location): ?>
                    <span><i class="fas fa-map-marker-alt"></i> <?php echo esc_html($location); ?></span>
                <?php endif; ?>
                <?php if ($capacity): ?>
                    <span><i class="fas fa-users"></i> <?php echo esc_html($capacity); ?></span>
                <?php endif; ?>
                <span class="badge <?php echo esc_attr($badge_class); ?>"><?php echo esc_html($status); ?></span>
            </div>
        </div>
        <a href="<?php echo esc_url($apply_url ?: $link); ?>"
           <?php echo $apply_url ? 'target="_blank" rel="noopener"' : ''; ?>
           class="schedule-btn">상세보기</a>
    </div>
    <?php
}

function parkgolf_title() {
    if (is_front_page())   { echo bloginfo('name'); }
    elseif (is_singular()) { the_title(); }
    elseif (is_archive())  { the_archive_title(); }
    elseif (is_search())   { echo '"' . get_search_query() . '" 검색 결과'; }
    else                   { echo bloginfo('name'); }
}

function parkgolf_schema_markup() {

// 자동 수집 카테고리: 파크골프뉴스(23), 대회(11), 소식(24), 협회(26)
define('PGK_NOINDEX_CATS', [23, 11, 24, 26]);

add_filter('rank_math/frontend/robots', function($robots) {
    // 자동 수집 카테고리 포스트 noindex
    if (is_singular('post')) {
        foreach (PGK_NOINDEX_CATS as $cat_id) {
            if (in_category($cat_id)) {
                unset($robots['index']);
                $robots['noindex'] = 'noindex';
                return $robots;
            }
        }
    }
    // 이벤트 커스텀 포스트 타입 전체 noindex
    if (is_singular('parkgolf_event')) {
        unset($robots['index']);
        $robots['noindex'] = 'noindex';
        return $robots;
    }
    // 자동 수집 카테고리 목록 페이지 + 이벤트 아카이브 noindex
    if (is_category(PGK_NOINDEX_CATS) || is_post_type_archive('parkgolf_event') || is_tax('event_type')) {
        unset($robots['index']);
        $robots['noindex'] = 'noindex';
        return $robots;
    }
    // 앱커뮤니티 카테고리 noindex (앱 전용)
    if (is_singular('post') && has_category('parkgolf-community')) {
        unset($robots['index']);
        $robots['noindex'] = 'noindex';
        return $robots;
    }
    return $robots;
}, 20);

function pgk_verify_kakao($kakao_token) {
    if (empty($kakao_token)) return null;
    $resp = wp_remote_get('https://kapi.kakao.com/v2/user/me', [
        'headers' => ['Authorization' => 'Bearer ' . $kakao_token],
        'timeout' => 10,
    ]);
    if (is_wp_error($resp)) return null;
    $body = json_decode(wp_remote_retrieve_body($resp), true);
    if (empty($body['id'])) return null;

    $kakao_id = (string) $body['id'];
    $nickname = $body['kakao_account']['profile']['nickname'] ?? '파크골퍼';
    $email    = $body['kakao_account']['email'] ?? ($kakao_id . '@kakao.parkgolf');
    $username = 'kakao_' . $kakao_id;
    $user     = get_user_by('login', $username);
    if (!$user) {
        $user_id = wp_create_user($username, wp_generate_password(32), $email);
        if (is_wp_error($user_id)) return null;
        wp_update_user(['ID' => $user_id, 'display_name' => $nickname]);
        $user = get_user_by('id', $user_id);
    } else {
        wp_update_user(['ID' => $user->ID, 'display_name' => $nickname]);
    }
    return $user;
}

function pgk_community_post(WP_REST_Request $request) {
    $kakao_token = sanitize_text_field($request->get_param('kakao_token'));
    $title       = sanitize_text_field($request->get_param('title'));
    $content     = wp_kses_post($request->get_param('content'));

    if (empty($title) || empty($content))
        return new WP_Error('missing_fields', '제목과 내용은 필수입니다.', ['status' => 400]);

    $user = pgk_verify_kakao($kakao_token);
    if (!$user)
        return new WP_Error('invalid_token', '카카오 인증 실패', ['status' => 401]);

    $cat    = get_category_by_slug('parkgolf-community');
    $cat_id = $cat ? $cat->term_id : 0;

    $post_id = wp_insert_post([
        'post_title'    => $title,
        'post_content'  => nl2br($content),
        'post_status'   => 'publish',
        'post_author'   => $user->ID,
        'post_category' => $cat_id ? [$cat_id] : [],
    ], true);

    if (is_wp_error($post_id))
        return new WP_Error('insert_failed', $post_id->get_error_message(), ['status' => 500]);

    return ['post_id' => $post_id, 'author' => $user->display_name];
}

function pgk_community_comment(WP_REST_Request $request) {
    $kakao_token = sanitize_text_field($request->get_param('kakao_token'));
    $post_id     = absint($request->get_param('post_id'));
    $content     = sanitize_textarea_field($request->get_param('content'));

    if (!$post_id || empty($content))
        return new WP_Error('missing_fields', '필수 값 누락', ['status' => 400]);

    $user = pgk_verify_kakao($kakao_token);
    if (!$user)
        return new WP_Error('invalid_token', '카카오 인증 실패', ['status' => 401]);

    $comment_id = wp_insert_comment([
        'comment_post_ID'      => $post_id,
        'comment_content'      => $content,
        'comment_author'       => $user->display_name,
        'comment_author_email' => $user->user_email,
        'user_id'              => $user->ID,
        'comment_approved'     => 1,
    ]);

    if (!$comment_id)
        return new WP_Error('comment_failed', '댓글 등록 실패', ['status' => 500]);

    return ['comment_id' => $comment_id, 'author' => $user->display_name];
}

add_action('init', function () {
    $uri = parse_url($_SERVER['REQUEST_URI'] ?? '', PHP_URL_PATH);
    if ($uri !== '/kakao-callback') return;

    $code = sanitize_text_field($_GET['code'] ?? '');
    if (!$code) {
        header('Location: parkgolfkkorea://oauth/kakao?error=no_code');
        exit;
    }

    // 인가 코드 → 액세스 토큰 교환
    $resp = wp_remote_post('https://kauth.kakao.com/oauth/token', [
        'body' => [
            'grant_type'   => 'authorization_code',
            'client_id'    => '83e317d4de45658d4be91a215af85150',
            'redirect_uri' => 'https://parkgolfk.com/kakao-callback',
            'code'         => $code,
        ],
        'timeout' => 10,
    ]);

    $access_token = '';
    if (!is_wp_error($resp)) {
        $data = json_decode(wp_remote_retrieve_body($resp), true);
        $access_token = $data['access_token'] ?? '';
    }

    if ($access_token) {
        $deeplink = 'parkgolfkkorea://oauth/kakao?token=' . urlencode($access_token);
    } else {
        $deeplink = 'parkgolfkkorea://oauth/kakao?error=token_failed';
    }
    // WebView의 onShouldStartLoadWithRequest가 parkgolfkkorea:// 를 인터셉트
    header('Content-Type: text/html; charset=utf-8');
    echo '<!DOCTYPE html><html><head><meta charset="utf-8">'
       . '<script>window.location.replace(' . json_encode($deeplink) . ');</script>'
       . '</head><body>'
       . '<p>앱으로 이동 중입니다... <a href="' . esc_attr($deeplink) . '">여기를 탭하세요</a></p>'
       . '</body></html>';
    exit;
});

// 얇은 콘텐츠 파크골프장 페이지 noindex
// 전화/이용요금/운영시간/주소 중 3개 이상 비어있으면 검색 제외
// ============================================================
function pgk_is_thin_course($id) {
    $empty = ['', '정보 없음', '현장 문의', '문의', '-', '해당 없음', '없음'];
    $score = 0;
    if (in_array(trim((string) (get_post_meta($id, 'course_phone',   true) ?: '')), $empty, true)) $score++;
    if (in_array(trim((string) (get_post_meta($id, 'course_fee',     true) ?: '')), $empty, true)) $score++;
    if (in_array(trim((string) (get_post_meta($id, 'course_hours',   true) ?: '')), $empty, true)) $score++;
    $addr = trim((string) (get_post_meta($id, 'course_address', true) ?: ''));
    if (empty($addr) || mb_strlen($addr) < 8) $score++;
    return $score >= 3;
}

// Rank Math 필터: index 제거 + noindex 추가 (기존 max-snippet 등 보존)
add_filter('rank_math/frontend/robots', function ($robots) {
    if (!is_singular('parkgolf_course')) return $robots;
    if (!pgk_is_thin_course(get_the_ID())) return $robots;
    unset($robots['index']);
    $robots['noindex'] = 'noindex';  // boolean 아닌 string — Rank Math 출력 형식 일치
    return $robots;
}, 20);

            'callback'            => 'pgk_get_scores',
            'permission_callback' => '__return_true',
        ],
    ]);
});

function pgk_save_score(WP_REST_Request $request) {
    $data        = $request->get_json_params();
    $kakao_token = sanitize_text_field($data['kakao_token'] ?? '');
    $user        = pgk_verify_kakao($kakao_token);

    $entry = [
        'id'          => 'srv_' . uniqid('', true),
        'course_name' => sanitize_text_field($data['course_name'] ?? ''),
        'holes'       => intval($data['holes'] ?? 9),
        'scores'      => array_map('intval', (array) ($data['scores'] ?? [])),
        'total_score' => intval($data['total_score'] ?? 0),
        'total_par'   => intval($data['total_par'] ?? 0),
        'date'        => sanitize_text_field($data['date'] ?? current_time('Y-m-d')),
        'saved_at'    => current_time('c'),
        'synced'      => true,
    ];

    if ($user) {
        $existing = get_user_meta($user->ID, 'pg_scores', true) ?: [];
        array_unshift($existing, $entry);
        if (count($existing) > 200) $existing = array_slice($existing, 0, 200);
        update_user_meta($user->ID, 'pg_scores', $existing);
    }

    return new WP_REST_Response(['success' => true, 'score' => $entry], 200);
}

function pgk_get_scores(WP_REST_Request $request) {
    $kakao_token = sanitize_text_field($request->get_param('kakao_token') ?? '');
    $user        = pgk_verify_kakao($kakao_token);

    if (!$user) {
        return new WP_REST_Response(['scores' => []], 200);
    }

    $scores = get_user_meta($user->ID, 'pg_scores', true) ?: [];
    return new WP_REST_Response(['scores' => array_values($scores)], 200);
}

add_action('rest_api_init', function () {

    ]);

    register_rest_route('parkgolf/v1', '/send-push-now', [
        'methods'             => 'POST',
        'callback'            => 'pgk_trigger_push_now',
        'permission_callback' => '__return_true',
    ]);
});

function pgk_trigger_push_now(WP_REST_Request $request) {
    // JSON body 또는 form data 모두 지원
    $secret = sanitize_text_field($request->get_param('secret') ?? '');
    // 시크릿 키는 wp-config.php 에 define('PGK_PUSH_SECRET_KEY', '...') 로 설정
    $expected = defined('PGK_PUSH_SECRET_KEY') ? PGK_PUSH_SECRET_KEY : '';
    if (empty($expected) || !hash_equals($expected, $secret)) {
        return new WP_REST_Response(['success' => false, 'message' => '인증 실패'], 403);
    }
    pgk_send_push_to_all();
    return new WP_REST_Response(['success' => true, 'time' => current_time('c')], 200);
}

function pgk_register_push_token(WP_REST_Request $request) {
    $data     = $request->get_json_params();
    $token    = sanitize_text_field($data['token'] ?? '');
    $platform = sanitize_text_field($data['platform'] ?? 'unknown');

    if (empty($token) || strpos($token, 'ExponentPushToken[') !== 0) {
        return new WP_REST_Response(['success' => false, 'message' => '유효하지 않은 토큰'], 400);
    }

    $tokens          = get_option('pgk_push_tokens', []);
    $tokens[$token]  = $platform;
    update_option('pgk_push_tokens', $tokens, false);

    return new WP_REST_Response(['success' => true], 200);
}

add_action('after_setup_theme', function () {

        // 다음 KST 08:00 = UTC 23:00 (전날)
        $next = strtotime('today 23:00:00 UTC');
        if ($next < time()) $next = strtotime('tomorrow 23:00:00 UTC');
        wp_schedule_event($next, 'pgk_8hours', 'pgk_push_cron');
    }
});

add_action('pgk_push_cron', 'pgk_send_push_to_all');

function pgk_send_push_to_all() {
    $tokens = get_option('pgk_push_tokens', []);
    if (empty($tokens)) return;

    // 최신 7건 중 랜덤 선택 (최신일수록 가중치 높음)
    $posts = get_posts([
        'post_type'   => ['post', 'parkgolf_event'],
        'post_status' => 'publish',
        'numberposts' => 7,
        'orderby'     => 'date',
        'order'       => 'DESC',
    ]);
    if (empty($posts)) return;

    // 가중 랜덤: 1번째=7, 2번째=6 ... 7번째=1
    $total = 0;
    $weights = [];
    foreach ($posts as $i => $_) {
        $w = max(1, 7 - $i);
        $weights[] = $w;
        $total += $w;
    }
    $rand = mt_rand(1, $total);
    $cum  = 0;
    $idx  = 0;
    foreach ($weights as $i => $w) {
        $cum += $w;
        if ($rand <= $cum) { $idx = $i; break; }
    }
    $post  = $posts[$idx];
    $type  = $post->post_type === 'parkgolf_event' ? 'event' : 'news';
    $title = get_the_title($post);
    $body  = wp_trim_words(wp_strip_all_tags(get_post_field('post_excerpt', $post->ID) ?: get_post_field('post_content', $post->ID)), 20, '...');

    $messages = [];
    foreach (array_keys($tokens) as $token) {
        $messages[] = [
            'to'    => $token,
            'title' => '파크골프코리아',
            'body'  => $title,
            'sound' => 'default',
            'data'  => ['postId' => $post->ID, 'type' => $type, 'title' => $title],
        ];
    }

    // Expo Push API: 최대 100건씩 배치 전송
    foreach (array_chunk($messages, 100) as $batch) {
        wp_remote_post('https://exp.host/--/api/v2/push/send', [
            'headers' => [
                'Content-Type' => 'application/json',
                'Accept'       => 'application/json',
            ],
            'body'    => wp_json_encode($batch),
            'timeout' => 15,
        ]);
    }
}
