<script>
const PER_PAGE    = 24;
let currentPage   = 1;
let currentRegion = '';
let currentView   = 'grid';
let debouncer     = null;
let totalPages    = 1;

document.addEventListener('DOMContentLoaded', () => {
  doSearch();

  document.getElementById('courseSearch').addEventListener('input', () => {
    clearTimeout(debouncer);
    debouncer = setTimeout(() => { currentPage = 1; doSearch(); }, 450);
  });
  document.getElementById('courseSearch').addEventListener('keydown', e => {
    if (e.key === 'Enter') { currentPage = 1; doSearch(); }
  });

  document.querySelectorAll('#regionChips .chip').forEach(btn => {
    btn.addEventListener('click', () => {
      document.querySelectorAll('#regionChips .chip').forEach(b => b.classList.remove('active'));
      btn.classList.add('active');
      currentRegion = btn.dataset.region;
      currentPage   = 1;
      doSearch();
    });
  });
});

function doSearch() {
  const keyword = document.getElementById('courseSearch').value.trim();
  const sort    = document.getElementById('sortSelect').value;
  const order   = sort === 'title' ? 'asc' : 'desc';

  let url = `<?php echo home_url(); ?>/wp-json/wp/v2/parkgolf_course?per_page=${PER_PAGE}&page=${currentPage}&_embed&orderby=${sort}&order=${order}`;
  if (keyword)       url += `&search=${encodeURIComponent(keyword)}`;
  if (currentRegion) url += `&region=${encodeURIComponent(currentRegion)}`;

  document.getElementById('courseGrid').innerHTML = '<div class="loading-msg">⛳ 검색 중...</div>';
  document.getElementById('searchPagination').innerHTML = '';

  fetch(url)
    .then(res => {
      const total = parseInt(res.headers.get('X-WP-Total') || '0');
      totalPages  = parseInt(res.headers.get('X-WP-TotalPages') || '1');
      return res.json().then(data => ({ data, total }));
    })
    .then(({ data, total }) => {
      renderResults(data, total);
      renderPagination();
    })
    .catch(() => {
      document.getElementById('courseGrid').innerHTML =
        '<div class="no-results"><div class="nr-icon">⚠️</div><h3>오류가 발생했습니다</h3><p>잠시 후 다시 시도해주세요.</p></div>';
    });
}

function renderResults(posts, total) {
  const grid = document.getElementById('courseGrid');
  document.getElementById('resultCount').textContent = `총 ${total.toLocaleString()}개 파크골프장`;

  if (!posts || !posts.length) {
    grid.innerHTML = `<div class="no-results">
      <div class="nr-icon">🔍</div>
      <h3>검색 결과가 없습니다</h3>
      <p>다른 키워드나 지역을 선택해보세요.</p>
    </div>`;
    return;
  }

  grid.className = 'course-grid' + (currentView === 'list' ? ' list-view' : '');
  grid.innerHTML = posts.map(post => {
    const title   = post.title?.rendered || '';
    const link    = post.link || '#';
    const address = post.meta?.course_address || '';
    const phone   = post.meta?.course_phone   || '';
    const fee     = post.meta?.course_fee     || '';
    const holes   = post.meta?.course_holes   || '';
    const hours   = post.meta?.course_hours   || '';
    const terms   = post._embedded?.['wp:term']?.[0] || [];
    const region  = terms[0]?.name || '';
    const isFree  = fee.includes('무료');
    const feeLabel = fee ? (isFree ? '무료' : '유료') : '현장문의';
    const feeClass = isFree ? 'free' : (fee ? 'paid' : 'inquiry');

    return `<div class="course-card">
      <a href="${link}" style="display:block;text-decoration:none;color:inherit;height:100%;">
        <div class="course-card-header">
          <span class="badge ${feeClass}">${feeLabel}</span>
          ${region ? `<span class="badge region">${region}</span>` : ''}
          <div class="course-img-placeholder">⛳</div>
        </div>
        <div class="course-card-body">
          <h3 class="course-title">${title}</h3>
          ${address ? `<p class="course-address"><i class="fas fa-map-marker-alt"></i> ${address}</p>` : ''}
          <div class="course-specs">
            ${holes ? `<span><i class="fas fa-golf-ball"></i> ${holes}홀</span>` : ''}
            ${hours ? `<span><i class="fas fa-clock"></i> ${hours}</span>` : ''}
            ${phone ? `<span><i class="fas fa-phone"></i> ${phone}</span>` : ''}
          </div>
        </div>
      </a>
    </div>`;
  }).join('');
}

function renderPagination() {
  const pag = document.getElementById('searchPagination');
  if (totalPages <= 1) { pag.innerHTML = ''; return; }

  let html = '';
  if (currentPage > 1)
    html += `<button class="page-btn" onclick="goPage(${currentPage-1})">◀ 이전</button>`;

  const start = Math.max(1, currentPage - 2);
  const end   = Math.min(totalPages, currentPage + 2);
  for (let i = start; i <= end; i++)
    html += `<button class="page-btn ${i===currentPage?'active':''}" onclick="goPage(${i})">${i}</button>`;

  if (currentPage < totalPages)
    html += `<button class="page-btn" onclick="goPage(${currentPage+1})">다음 ▶</button>`;

  pag.innerHTML = html;
}

function goPage(p) {
  currentPage = p;
  doSearch();
  window.scrollTo({ top: 0, behavior: 'smooth' });
}

function setView(v) {
  currentView = v;
  document.querySelectorAll('.view-btn').forEach((b, i) => {
    b.classList.toggle('active', (v==='grid'&&i===0)||(v==='list'&&i===1));
  });
  const g = document.getElementById('courseGrid');
  g.classList.toggle('list-view', v === 'list');
}
</script>
